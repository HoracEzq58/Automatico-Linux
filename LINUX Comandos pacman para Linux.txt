Script ChatGPT del 10/04/2025

#!/bin/bash

echo "🔧 Instalando utilidades necesarias..."
sudo apt update
sudo apt install -y cpufrequtils lm-sensors psensor

echo "🔍 Detectando sensores (esto puede tardar un poquito)..."
yes | sudo sensors-detect > /dev/null

echo "📦 Detectando gobernadores de CPU disponibles..."
AVAILABLE=$(cpufreq-info -g)

if [ -z "$AVAILABLE" ]; then
    echo "❌ No se detectaron gobernadores. ¿Tu CPU lo soporta?"
    exit 1
fi

echo "✅ Gobernadores disponibles: $AVAILABLE"
echo
read -rp "📌 Elegí uno (copialo tal como aparece): " GOVERNOR

if [[ ! "$AVAILABLE" =~ $GOVERNOR ]]; then
    echo "❌ Gobernador no válido. Abortando."
    exit 1
fi

echo "⚙️ Aplicando '$GOVERNOR' a todos los núcleos..."
for cpu in /sys/devices/system/cpu/cpu[0-9]*; do
    echo $GOVERNOR | sudo tee "$cpu/cpufreq/scaling_governor" > /dev/null
done

echo "🛠️ Creando servicio systemd para usar '$GOVERNOR' al inicio..."

sudo tee /etc/systemd/system/cpufreq-custom.service > /dev/null <<EOF
[Unit]
Description=Set CPU governor to $GOVERNOR
After=multi-user.target

[Service]
Type=oneshot
ExecStart=/usr/bin/cpufreq-set -r -g $GOVERNOR

[Install]
WantedBy=multi-user.target
EOF

echo "📎 Habilitando servicio..."
sudo systemctl daemon-reload
sudo systemctl enable cpufreq-custom.service
sudo systemctl start cpufreq-custom.service

echo "📂 Agregando Psensor al inicio automático..."

mkdir -p ~/.config/autostart

cat > ~/.config/autostart/psensor.desktop <<EOF
[Desktop Entry]
Type=Application
Exec=psensor
Hidden=false
NoDisplay=false
X-GNOME-Autostart-enabled=true
Name=Psensor
Comment=Monitorea temperatura del CPU
EOF

echo
echo "✅ Todo listo:"
echo "🌡️ Psensor mostrará la temperatura en la barra"
echo "⚡ El CPU usará '$GOVERNOR' automáticamente al encender"
echo
echo "🔥 Podés verificar con 'sensors' y ajustar Psensor desde su ícono en la barra"
